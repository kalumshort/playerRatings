// src/app/robots.ts
import { MetadataRoute } from 'next';

export default function robots(): MetadataRoute.Robots {
  return {
    rules: {
      userAgent: '*',
      allow: '/',
      disallow: ['/profile/', '/admin/'], // Protect user-specific routes
    },
    sitemap: 'https://11votes.com/sitemap.xml',
  };
}
